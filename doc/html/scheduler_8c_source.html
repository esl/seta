66 void stack_depth_computation(processor_t *proc, closure_t *closure, seta_context_t *context) {
67  msg_list_t *allocated_ancient_list = (msg_list_t *)context->allocated_ancient_list;
68  if (seta_graph) {

71  }
72  closure->allocated_ancients = context->allocated_ancients;
73  int c1 = closure_space(closure);

75  int S1_candidate = c1 + c2;
76  if (S1_candidate > S1) {

313 void *scheduler_scheduling_loop(void *ptr) {
314  int n_local_proc = *(int *)ptr;
315  processor_t *local_proc = processors[n_local_proc];
316  while (1) {
317  processor_lock_ready_queue(local_proc);
318  closure_t *cl = ready_queue_extract_head_from_deepest_level(local_proc->rq);
319  processor_unlock_ready_queue(local_proc);
320  if (cl != NULL) {
321  scheduler_execute_closure(local_proc, cl);
322 
323  if (stop_computation) {
324  break;
325  }
326  }
327  else {
328  if (stop_computation) {
329  break;
330  }
331  int n_rand_proc = 0;
332  do {
333  n_rand_proc = rand() % n_proc;
334  } while (n_rand_proc == n_local_proc);
335 
336  processor_t *remote_proc = processors[n_rand_proc];
337 
338  processor_lock_ready_queue(remote_proc);
339  closure_t *rcl = ready_queue_extract_tail_from_shallowest_level(remote_proc->rq);
340  processor_unlock_ready_queue(remote_proc);
341 
342  if (rcl != NULL) {
343  scheduler_execute_closure(local_proc, rcl);
344  }
345  }
346  }
347  return NULL;
348 }

361 int seta_start(void *fun, int n) {
362  n_proc = n;
363  time_t t;
364  srand((unsigned)time(&t));
365 
366  if (seta_graph) {
367  graph_start(graph_filename);
368  }
369 
370  if (!(n_proc > 0)) {
371  printf("the number of processors must be greater than 0\n");
372  return -1;
373  }
374  processors = (processor_t **)malloc(sizeof(processor_t *) * n_proc);
375 
376  for (int i=0; i<n_proc; i++) {
377  processors[i] = processor_create(i);
378  if (seta_info) {
379  processor_create_info(processors[i]);
380  }
381  }
382 
383  closure_t *closure = closure_create();
384  if (seta_info) {
385  closure_create_info(closure);
386  }
387  if (seta_graph) {
388  closure_create_graph(closure, "entry");
389  closure->allocated_ancient_list = msg_list_create();
390  }
391  closure_set_fun(closure, fun);
392  closure->is_first_thread = true;
393 
394  processor_lock_ready_queue(processors[0]);
395  ready_queue_post_closure_to_level(processors[0]->rq, closure, 0);
396  processor_unlock_ready_queue(processors[0]);
397 
398  if (seta_info) {
399  scheduler_computate_processor_space(processors[0]);
400  }
401 
402  for (int i=0; i<n_proc; i++) {
403  processor_start(processors[i]);
404  }
405  for (int i=0; i<n_proc; i++) {
406  processor_wait_4_me(processors[i]);
407  }
408 
409  if (seta_info) {
410  scheduler_print_results();
411  }
412 
413  for (int i=0; i<n_proc; i++) {
414  if (seta_info) {
415  processor_destroy_info(processors[i]);
416  }
417  processor_destroy(processors[i]);
418  }
419 
420  free(processors);
421 
422  if (seta_graph && seta_info) {
423  msg_list_foreach(&graph_color_S1_element, S1_list);
424  msg_list_destroy(S1_list);
425  }
426  if (seta_graph) {
427  graph_finish();
428  }
429 
430  return 0;
431 }
